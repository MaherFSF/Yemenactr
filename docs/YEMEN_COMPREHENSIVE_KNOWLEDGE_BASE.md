-- Organizations table
CREATE TABLE organizations (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(255) NOT NULL,
  nameAr VARCHAR(255),
  acronym VARCHAR(50),
  type ENUM('un', 'ingo', 'ngo', 'government', 'donor', 'bank', 'company', 'research', 'media'),

CREATE TABLE data_sources (
  id INT PRIMARY KEY AUTO_INCREMENT,
